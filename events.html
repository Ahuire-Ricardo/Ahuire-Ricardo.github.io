HTML DOM events allow JavaScript to register different event handlers on elements in an HTML document.

Events are normally used in combination with functions, and the function will not be executed before the event occurs (such as when a user clicks a button).

onTouch

This program change the position by box with touch screen.


window.addEventListener('load', function(){

var boxmove = document.getElementById('boxmove'),
boxleft, 
startx, 
dist = 0, 
touchobj = null

boxmove.addEventListener('touchstart', function(e){
		touchobj = e.changedTouches[0] // reference first touch point
		boxleft = parseInt(boxmove.style.left) // get left position of box
		startx = parseInt(touchobj.clientX) // get x coord of touch point
		e.preventDefault() // prevent default click behavior
}, false)

boxmove.addEventListener('touchmove', function(e){
		touchobj = e.changedTouches[0] // reference first touch point for this event
		var dist = parseInt(touchobj.clientX) - startx // calculate dist traveled by touch point
		// move box according to starting pos plus dist
		// with lower limit 0 and upper limit 350 so it doesn't move outside box:
		boxmove.style.left = ( (boxleft + dist > 350)? 350 : (boxleft + dist < 0)? 0 : boxleft + dist ) + 'px'
		e.preventDefault()
}, false)

}, false)

 function changeBg()

    document.getElementById("body").style.background = "red";

function dblClick()
{
    document.getElementById("body").style.color = "black";
    document.getElementById("body").innerHTML = "Double Click!";
}
function contMenu()
{
    document.getElementById("body").style.color = "black";
    document.getElementById("body").style.background = "white";
}

When you click, the background will change to red and it will tell you to let go.
When you double click, the text color will change back to white and the text will change back to "Click".
